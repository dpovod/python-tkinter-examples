Програма Hello World!

Ось приклад коду для створення вікна повідомлення за допомогою Tkinter:


import tkinter as tk
from tkinter import messagebox

# Функція для відображення вікна повідомлення
def show_message():
    messagebox.showinfo("Відповідь", "Hello Again!")

# Створення головного вікна
root = tk.Tk()
root.title("Hello World!")

# Текст на вікні
label = tk.Label(root, text="Hello World!")
label.pack(pady=10)

# Кнопка
button = tk.Button(root, text="Привіт", command=show_message)
button.pack(pady=10)

# Запуск програми
root.mainloop()
